C语言计算文件大小

by YangYang
in template on 2017-09-21 | tags: 文件大小计算 
static int file_size(char fileFullPath[])
{
        long fileSize = 0;
        FILE *pFile = fopen(fileFullPath,"r");
        if( pFile == NULL)
        {
                printf("Open File %s Error /n", fileFullPath);
                return -1;
        }
        else
        {
                fseek(pFile, 0, SEEK_END);
                fileSize = ftell(pFile);
        }

        fclose(pFile);
        return fileSize;
}
